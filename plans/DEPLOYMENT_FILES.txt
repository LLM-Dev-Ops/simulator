QUICK DEPLOYMENT COMMANDS:

# Local Development
./llm-simulator serve                          # Single binary
docker-compose up -d                           # Full stack

# Kubernetes
make helm-install                              # Helm deployment
kubectl apply -f deploy/kubernetes/            # Raw manifests

# Verification
make health                                    # Health check
make k8s-status                                # Deployment status
make metrics                                   # Prometheus metrics

# Scaling
kubectl scale deployment llm-simulator --replicas=10

# Monitoring
make k8s-logs                                  # View logs
kubectl top pods -n llm-devops                 # Resource usage
